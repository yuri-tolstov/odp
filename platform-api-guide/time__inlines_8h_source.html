37 static inline odp_time_t _odp_time_cur(void)
38 {
39  if (_odp_time_glob.use_hw) {
40  odp_time_t time;
41 
42  time.count = _odp_cpu_global_time() - _odp_time_glob.hw_start;
43  return time;
44  }
45 
46  return _odp_timespec_cur();
47 }

49 static inline uint64_t _odp_time_hw_to_ns(odp_time_t time)
50 {
51  uint64_t nsec;
52  uint64_t freq_hz = _odp_time_glob.hw_freq_hz;
53  uint64_t count = time.count;
54  uint64_t sec = 0;
55 
56  if (count >= freq_hz) {
57  sec = count / freq_hz;
58  count = count - sec * freq_hz;
59  }
60 
61  nsec = (_ODP_TIME_GIGA_HZ * count) / freq_hz;
62 
63  return (sec * _ODP_TIME_GIGA_HZ) + nsec;
64 }

66 static inline uint64_t _odp_time_convert_to_ns(odp_time_t time)
67 {
68  if (_odp_time_glob.use_hw)
69  return _odp_time_hw_to_ns(time);
70 
71  return time.nsec;
72 }

115 _ODP_INLINE int odp_time_cmp(odp_time_t t2, odp_time_t t1)
116 {
117  if (odp_likely(t2.u64 > t1.u64))
118  return 1;
119 
120  if (t2.u64 < t1.u64)
121  return -1;
122 
123  return 0;
124 }
